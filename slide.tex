\begin{frame}{Introduction}
    \begin{itemize}
        \item Let $N$ be the population in 2021 = 70,000,000 people.
        
        \item Let $\gamma$ be the recovery rate: 
        $$ \gamma = 1/\text{recovery days} $$
        Since recovery $\approx 8$ days $\Rightarrow \gamma = 1/8 = 0.125$.
        
        \item Let $\beta$ be the infection rate (speed of spread) $\rightarrow R_t = \beta / \gamma$.
        
        \item To find the best $\beta$, I used SSE to minimize the error:
        $$ \text{SSE}(\beta) = \sum_{t=0}^{n} (I_{\text{real}}(t) - I_{\text{model}}(t,\beta))^2 $$
        
        \textbf{Where:}
        \begin{itemize}
            \small 
            \item \textbf{SSE($\beta$):} Total sum of squared errors.
            \item $I_{\text{real}}(t)$: Number of infected people on day $t$ (Red dot).
            \item $I_{\text{model}}(t,\beta)$: Number of infected people predicted by the equation on day $t$ (Blue line).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Data}

\begin{frame}{Result}
    \begin{itemize}

    \end{itemize}
